Create a PDBPolicy to get started:

  cat <<EOF | kubectl apply -f -
  apiVersion: availability.pdboperator.io/v1alpha1
  kind: PDBPolicy
  metadata:
    name: my-pdb-policy
  spec:
    targetReference:
      kind: Deployment
    pdbSpec:
      minAvailable: "50%"
    selector:
      matchLabels:
        app: my-app
  EOF
